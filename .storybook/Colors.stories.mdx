import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs/blocks";

import { dark, light } from "../src/styles/theme";

<Meta title="UI/Colors" />

<style>
  {`
  .colors {
    display: flex;
    padding: 10px;
  }
  .color {
    padding: 20px;
    margin: 0 40px;
  }`}
</style>

export const getColors = (theme) =>
  Object.keys(theme).map((key) => ({ name: key, value: theme[key] }));

# Colors

<div className="colors">
  <div id="dark" title="Dark mode" className="color">
    <div>Dark mode</div>
    <ColorPalette>
      {getColors(dark).map(({ name, value }) => (
        <ColorItem
          key={name}
          title={`theme.${name}`}
          subtitle={name}
          colors={[value]}
        />
      ))}
    </ColorPalette>
  </div>
  <div id="light" title="Light mode" className="color">
    <div>Light mode</div>
    <ColorPalette>
      {getColors(light).map(({ name, value }) => (
        <ColorItem
          key={name}
          title={`theme.${name}`}
          subtitle={name}
          colors={[value]}
        />
      ))}
    </ColorPalette>
  </div>
</div>
